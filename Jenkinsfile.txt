pipeline {
    
    agent any
    def customimage
   
stages{
	stage('Build'){
            steps {
		sh 'mvn -f HelloWorld/pom.xml clean package'
            }
            post {
                success {
                    echo 'Now Archiving...'
                    archiveArtifacts artifacts: '**/target/*.jar'
                }
            }
        }

        stage('buildimage')
	{
            steps 
		{
                    sh 'customimage = docker build .'
		    sh 'customimage.push()'
            	}            
        }        
        }
    
}
