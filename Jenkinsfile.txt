pipeline {
    agent any 
   
stages{
	stage('Build'){
            steps {
		sh 'mvn -f HelloWorld/pom.xml clean package'
            }
            post {
                success {
                    echo 'Now Archiving...'
                    archiveArtifacts artifacts: '**/target/*.jar'
                }
            }
        }

        stage('buildimage')
	{
            steps 
		{
                    def customimage = docker build -t myfirstimage:1 .
		    docker.withRegistry('https://registry.hub.docker.com');
		    customimage.push();
            	}            
        }        
        }
    
}
