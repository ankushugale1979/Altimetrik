pipeline {
    agent any 
   
stages{
	stage('Build'){
            steps {
		sh 'mvn -f HelloWorld/pom.xml clean package'
            }
            post {
                success {
                    echo 'Now Archiving...'
                    archiveArtifacts artifacts: '**/target/*.jar'
                }
            }
        }

        stage('buildimage')
	{
            steps 
		{
                    sh 'def customimage = docker build -t myfirstimage:1 .'
		    sh 'docker.withRegistry('https://registry.hub.docker.com')'
		    sh 'customimage.push()'
            	}            
        }        
        }
    
}
