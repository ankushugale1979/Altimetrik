pipeline {
    agent any 
   
stages{
	stage('Build'){
            steps {
                sh 'cd HelloWorld'
		sh 'mvn clean package'
            }
            post {
                success {
                    echo 'Now Archiving...'
                    archiveArtifacts artifacts: '**/target/*.war'
                }
            }
        }

        stage('buildimage')
	{
            steps 
		{
                sh 'docker build .'
            	}            
        }        
        }
    
}
